// src/assets/js/print-bridge.js
const isTauri = !!window.__TAURI__
const isCapacitor = !!window.Capacitor

export async function printPdfNative(pdfBlob) {
  if (isTauri) {
    const { invoke } = window.__TAURI__.core
    // Save blob ke temp file, lalu invoke print_pdf
    const buffer = await pdfBlob.arrayBuffer()
    const tempPath = await invoke('save_temp_pdf', { buffer: Array.from(new Uint8Array(buffer)) })
    await invoke('print_pdf', { pdfPath: tempPath })
  } else if (isCapacitor) {
    // Capacitor → buka share sheet / open with PDF viewer
    // (Print direct di Android perlu plugin tambahan)
    const url = URL.createObjectURL(pdfBlob)
    window.open(url, '_blank')
  } else {
    // Web → window.print() atau download
    const url = URL.createObjectURL(pdfBlob)
    const iframe = document.createElement('iframe')
    iframe.style.display = 'none'
    iframe.src = url
    iframe.onload = () => iframe.contentWindow.print()
    document.body.appendChild(iframe)
  }
}

```rust
use std::process::Command;

#[tauri::command]
pub fn print_pdf_silent(pdf_path: String, printer: Option<String>) -> Result<(), String> {
    #[cfg(target_os = "windows")]
    {
        // PowerShell Out-Printer untuk silent print
        let mut cmd = Command::new("powershell");
        cmd.arg("-Command");
        if let Some(p) = printer {
            cmd.arg(format!(
                "Start-Process -FilePath '{}' -Verb PrintTo -ArgumentList '\"{}\"'",
                pdf_path, p
            ));
        } else {
            cmd.arg(format!(
                "Start-Process -FilePath '{}' -Verb Print",
                pdf_path
            ));
        }
        cmd.spawn().map_err(|e| e.to_string())?;
    }
    Ok(())
}

#[tauri::command]
pub fn list_printers() -> Result<Vec<String>, String> {
    #[cfg(target_os = "windows")]
    {
        let output = Command::new("powershell")
            .args(&["-Command", "Get-Printer | Select-Object -ExpandProperty Name"])
            .output()
            .map_err(|e| e.to_string())?;
        let stdout = String::from_utf8_lossy(&output.stdout);
        let printers: Vec<String> = stdout.lines().map(|s| s.trim().to_string()).collect();
        Ok(printers)
    }
    #[cfg(not(target_os = "windows"))]
    {
        Ok(vec![])
    }
}
```
